def collatz_key_generator(seed, length):
    """
    Collatz varsayımını kullanarak belirli uzunlukta bir bit dizisi üretir.
    """
    key = ""
    current_n = seed
    
    while len(key) < length:
        if current_n % 2 == 0:
            current_n = current_n // 2
            key += "0"
        else:
            current_n = 3 * current_n + 1
            key += "1"
            
        # Collatz 4-2-1 döngüsüne girerse rastgeleliği korumak için n'i değiştiriyoruz
        if current_n == 1:
            current_n = seed + len(key) + 7 

    return key

# Kullanıcıdan girdi alma
try:
    s = int(input("Başlangıç sayısı (Seed) girin: "))
    l = int(input("Üretilecek anahtar bit uzunluğu: "))
    
    generated_key = collatz_key_generator(s, l)
    
    print("\n--- Üretilen Anahtar ---")
    print(f"Seed: {s}")
    print(f"Bit Dizisi: {generated_key}")
    print(f"Hex Formatı: {hex(int(generated_key, 2))[2:].upper()}")
except ValueError:
    print("Lütfen geçerli tam sayılar girin.")
